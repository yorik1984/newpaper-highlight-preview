 3 local util         = require("newpaper.util")
 4 local configModule = require("newpaper.config")

 6 local M = {}
 7 
 8 function util.syntax(syntax)
 9     for group, colors in pairs(syntax) do
10         util.highlight(group, colors)
11     end
12 end
13 
14 function util.loadCustomSyntax(config)
15     local async
16     async = vim.loop.new_async(vim.schedule_wrap(function ()
17         if type(config.custom_highlights) == 'table' then
18             util.syntax(config.custom_highlights)
19         end
20         async:close()
21     end))
22     async:send()
23 end
24 
25 function util.onColorScheme()
26     if vim.g.colors_name ~= "newpaper" then
27         vim.cmd([[autocmd! newpaper]])
28         vim.cmd([[augroup! newpaper]])
29     end
30 end

32 local function opt(key, default)
33     key = "newpaper_" .. key
34     if vim.g[key] == nil then
35         return default
36     end
37 end

42     table_option        = opt("sidebars", {}),
43     number              = 1234567890,

47 function M.setup(config)
48     config = config or configModule.config
49     local newpaper = {
50         black           = '#2B2B2B', -- color00
51         maroon          = '#AF0000', -- color01
52         darkgreen       = '#008700'  -- color02
53     }
54     local syntax = {
55         StorageClass   = { fg = newpaper.ocean },
56         Structure      = { fg = newpaper.darkpurple },
57         Conditional    = { fg = newpaper.keyword }
58     }
59     if vim.g.newpaper_colors == nil then
60         vim.g.newpaper_colors = configModule.config.colors
61     end
62     util.color_overrides(newpaper, config)
63     return newpaper

65 end
